## 题目描述

用两个栈来实现一个队列，完成队列的Push和Pop操作。 队列中的元素为int类型。
    
    

```python
    class Solution
    {
    public:
        void push(int node) {
           while(!stack2.empty())//入队时要保证 Stack2为空
           {
               stack1.push(stack2.top());
               stack2.pop();
           }
            stack1.push(node);
        }
        int pop() {
            while(!stack1.empty())//入队时要保证 Stack1为空
           {
               stack2.push(stack1.top());
               stack1.pop();
           }
           int temp = stack2.top();
            stack2.pop();
            return temp;
        }
    private:
        stack<int> stack1;
        stack<int> stack2;
    };
```

  
运行时间：3ms  
占用内存：460K
