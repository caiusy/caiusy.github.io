## 题目描述

    /**  
    *  struct ListNode {  
    *        int val;  
    *        struct ListNode *next;  
    *        ListNode(int x) :  
    *              val(x), next(NULL) {  
    *        }  
    *  };  
    */  
    class Solution {  
    public:  
        vector<int> printListFromTailToHead(ListNode* head) {  
            vector<int> value;  
            ListNode *p=NULL;  
            p=head;  
            stack<int> stk;  
            while(p!=NULL){  
                stk.push(p->val);  
                p=p->next;  
            }  
            while(!stk.empty()){  
                value.push_back(stk.top());  
                stk.pop();  
            }  
            return value;  
        }    
    };
