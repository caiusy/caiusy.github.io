    /*  
    struct ListNode {  
    	int val;  
    	struct ListNode *next;  
    	ListNode(int x) :  
    			val(x), next(NULL) {  
    	}  
    };*/  
    class Solution {  
    public:  
        ListNode* ReverseList(ListNode* pHead) {  
            ListNode *prev = NULL;  
            ListNode *curr = pHead;  
            ListNode *next = NULL;  
            while(curr!=NULL)  
            {  
                next = curr->next;  
                curr->next = prev;  
                prev = curr;  
                curr = next;  
            }  
            return prev;  
        }  
    };
